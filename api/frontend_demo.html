🛠️ JavaScript SDK 使用示例

基本用法

// 引入SDK const automata = new AutoMATA_SDK('http://localhost:8000/api'); // 获取可用模型 const models = await automata.models.getAvailable(); // 创建训练任务 const task = await automata.training.create({ taskName: '我的训练任务', modelType: 'mlp', parameters: { epochs: 10, batch_size: 32 }, datasetPath: '/data/my_data.csv', userId: 1 }); // 监控任务状态 const stopMonitoring = automata.training.monitorStatus( taskId, (status, task) => { console.log('任务状态:', status); if (status === 'completed') { console.log('训练完成!'); } } );

Vue.js 集成

// 在Vue组件中使用 export default { data() { return { taskStatus: 'pending', logs: [] }; }, methods: { async createTask() { try { const task = await this.$automata.training.create({ taskName: this.taskName, modelType: this.modelType, parameters: this.parameters, userId: this.userId }); // 监控任务状态 this.$automata.training.monitorStatus( task.id, (status) => { this.taskStatus = status; } ); } catch (error) { console.error('创建任务失败:', error); } } } };
